1. 模拟space-between和间隙

      .container {
        display: flex;
        width: 600px;
        flex-wrap: wrap;
        flex-direction: row;
      }
      .item {
        background: skyblue;
        width: 24%;
        height: 150px;
        margin-top: 15px;
      }
      .item:not(:nth-last-child(4n)) {
        margin-right: calc(4% / 3);
      }
      html：
      <div class="container">
        <div class="item">
        <div class="item">
        <div class="item">
        <div class="item">
        <div class="item">
        <div class="item">
        <div class="item">
      </div>

2. 根据个数最后一个元素动态margin

    .container {
      display: flex;
      width: 600px;
      flex-wrap: wrap;
      flex-direction: row;
      justify-content: space-between;
    }
    /* 如果最后一行是3个元素 */
    .item:last-child:nth-child(4n - 1) {
        margin-right: calc(24% + 4% / 3);
    }
    /* 如果最后一行是2个元素 */
    .item:last-child:nth-child(4n - 2) {
        margin-right: calc(48% + 8% / 3);
    }
